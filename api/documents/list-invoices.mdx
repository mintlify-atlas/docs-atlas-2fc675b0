---
title: "List Invoices"
api: "GET /api/v1/invoices"
---

Retrieve a paginated list of invoices (facturas) for the authenticated user's company.

## Authentication

Requires a valid Bearer token with invoice viewing permissions.

```bash
Authorization: Bearer YOUR_TOKEN_HERE
```

## Query Parameters

<ParamField query="page" type="integer" default="1">
  Page number for pagination
</ParamField>

<ParamField query="per_page" type="integer" default="20">
  Number of invoices per page (max 100)
</ParamField>

<ParamField query="company_id" type="integer">
  Filter by specific company ID
</ParamField>

<ParamField query="branch_id" type="integer">
  Filter by specific branch ID
</ParamField>

<ParamField query="client_id" type="integer">
  Filter by specific client ID
</ParamField>

<ParamField query="serie" type="string">
  Filter by document series (e.g., "F001")
</ParamField>

<ParamField query="fecha_inicio" type="date">
  Start date filter (YYYY-MM-DD format)
</ParamField>

<ParamField query="fecha_fin" type="date">
  End date filter (YYYY-MM-DD format)
</ParamField>

<ParamField query="estado_sunat" type="string">
  Filter by SUNAT status: `pendiente`, `aceptado`, `rechazado`
</ParamField>

## Response

<ResponseField name="data" type="array">
  Array of invoice objects
  
  <Expandable title="invoice properties">
    <ResponseField name="id" type="integer">
      Invoice unique identifier
    </ResponseField>
    
    <ResponseField name="company_id" type="integer">
      Company that issued the invoice
    </ResponseField>
    
    <ResponseField name="branch_id" type="integer">
      Branch that issued the invoice
    </ResponseField>
    
    <ResponseField name="client_id" type="integer">
      Client receiving the invoice
    </ResponseField>
    
    <ResponseField name="tipo_documento" type="string">
      Document type code ("01" for invoice)
    </ResponseField>
    
    <ResponseField name="serie" type="string">
      Document series (e.g., "F001")
    </ResponseField>
    
    <ResponseField name="correlativo" type="string">
      Sequential number within the series
    </ResponseField>
    
    <ResponseField name="numero_completo" type="string">
      Full document number (e.g., "F001-00000123")
    </ResponseField>
    
    <ResponseField name="fecha_emision" type="date">
      Issue date
    </ResponseField>
    
    <ResponseField name="moneda" type="string">
      Currency code (PEN, USD)
    </ResponseField>
    
    <ResponseField name="mto_imp_venta" type="number">
      Total amount including taxes
    </ResponseField>
    
    <ResponseField name="estado_sunat" type="string">
      SUNAT status: pendiente, aceptado, rechazado
    </ResponseField>
    
    <ResponseField name="sunat_respuesta" type="string" nullable>
      Response message from SUNAT
    </ResponseField>
    
    <ResponseField name="client" type="object">
      Client information (nombre_comercial, numero_documento)
    </ResponseField>
    
    <ResponseField name="company" type="object">
      Company information
    </ResponseField>
    
    <ResponseField name="branch" type="object">
      Branch information
    </ResponseField>
  </Expandable>
</ResponseField>

<ResponseField name="current_page" type="integer">
  Current page number
</ResponseField>

<ResponseField name="per_page" type="integer">
  Items per page
</ResponseField>

<ResponseField name="total" type="integer">
  Total number of invoices matching the filters
</ResponseField>

<ResponseField name="last_page" type="integer">
  Last page number
</ResponseField>

## Code Examples

<CodeGroup>

```bash cURL
curl -X GET "https://your-domain.com/api/v1/invoices?page=1&per_page=20&estado_sunat=aceptado" \
  -H "Authorization: Bearer YOUR_TOKEN_HERE" \
  -H "Accept: application/json"
```

```javascript JavaScript
const response = await fetch(
  'https://your-domain.com/api/v1/invoices?page=1&per_page=20&estado_sunat=aceptado',
  {
    headers: {
      'Authorization': 'Bearer YOUR_TOKEN_HERE',
      'Accept': 'application/json'
    }
  }
);

const data = await response.json();
console.log(`Found ${data.total} invoices`);
```

```php PHP
$ch = curl_init();
curl_setopt($ch, CURLOPT_URL, 'https://your-domain.com/api/v1/invoices?page=1&per_page=20');
curl_setopt($ch, CURLOPT_RETURNTRANSFER, true);
curl_setopt($ch, CURLOPT_HTTPHEADER, [
    'Authorization: Bearer YOUR_TOKEN_HERE',
    'Accept: application/json'
]);

$response = curl_exec($ch);
$data = json_decode($response, true);
curl_close($ch);
```

</CodeGroup>
