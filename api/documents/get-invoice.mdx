---
title: "Get Invoice"
api: "GET /api/v1/invoices/{id}"
---

Retrieve detailed information for a specific invoice including all line items, tax calculations, and SUNAT status.

## Authentication

Requires a valid Bearer token with invoice viewing permissions.

```bash
Authorization: Bearer YOUR_TOKEN_HERE
```

## Path Parameters

<ParamField path="id" type="integer" required>
  The unique identifier of the invoice to retrieve
</ParamField>

## Response

<ResponseField name="id" type="integer">
  Invoice unique identifier
</ResponseField>

<ResponseField name="company_id" type="integer">
  Company that issued the invoice
</ResponseField>

<ResponseField name="branch_id" type="integer">
  Branch that issued the invoice
</ResponseField>

<ResponseField name="client_id" type="integer">
  Client receiving the invoice
</ResponseField>

<ResponseField name="tipo_documento" type="string">
  Document type code ("01" for factura)
</ResponseField>

<ResponseField name="serie" type="string">
  Document series (e.g., "F001")
</ResponseField>

<ResponseField name="correlativo" type="string">
  Sequential number
</ResponseField>

<ResponseField name="numero_completo" type="string">
  Full document number (serie-correlativo)
</ResponseField>

<ResponseField name="fecha_emision" type="date">
  Issue date (YYYY-MM-DD)
</ResponseField>

<ResponseField name="fecha_vencimiento" type="date" nullable>
  Due date for payment
</ResponseField>

<ResponseField name="moneda" type="string">
  Currency code (PEN, USD)
</ResponseField>

<ResponseField name="tipo_operacion" type="string">
  Operation type code (0101=Venta interna, 0200=Exportación, etc.)
</ResponseField>

<ResponseField name="mto_oper_gravadas" type="number">
  Taxable operations amount
</ResponseField>

<ResponseField name="mto_igv" type="number">
  IGV (tax) amount
</ResponseField>

<ResponseField name="mto_imp_venta" type="number">
  Total sale amount including taxes
</ResponseField>

<ResponseField name="estado_sunat" type="string">
  SUNAT status: pendiente, aceptado, rechazado
</ResponseField>

<ResponseField name="detalles" type="array">
  Array of invoice line items with quantities, prices, and taxes
</ResponseField>

<ResponseField name="client" type="object">
  Complete client information
  
  <Expandable title="client properties">
    <ResponseField name="id" type="integer">Client ID</ResponseField>
    <ResponseField name="tipo_documento" type="string">Document type (6=RUC)</ResponseField>
    <ResponseField name="numero_documento" type="string">RUC number</ResponseField>
    <ResponseField name="nombre_comercial" type="string">Commercial name</ResponseField>
    <ResponseField name="direccion" type="string">Address</ResponseField>
  </Expandable>
</ResponseField>

<ResponseField name="company" type="object">
  Complete company information
</ResponseField>

<ResponseField name="branch" type="object">
  Complete branch information
</ResponseField>

## Code Examples

<CodeGroup>

```bash cURL
curl -X GET https://your-domain.com/api/v1/invoices/123 \
  -H "Authorization: Bearer YOUR_TOKEN_HERE" \
  -H "Accept: application/json"
```

```javascript JavaScript
const invoiceId = 123;
const response = await fetch(
  `https://your-domain.com/api/v1/invoices/${invoiceId}`,
  {
    headers: {
      'Authorization': 'Bearer YOUR_TOKEN_HERE',
      'Accept': 'application/json'
    }
  }
);

const invoice = await response.json();
console.log(invoice.numero_completo);
```

```php PHP
$invoiceId = 123;
$ch = curl_init();
curl_setopt($ch, CURLOPT_URL, "https://your-domain.com/api/v1/invoices/{$invoiceId}");
curl_setopt($ch, CURLOPT_RETURNTRANSFER, true);
curl_setopt($ch, CURLOPT_HTTPHEADER, [
    'Authorization: Bearer YOUR_TOKEN_HERE',
    'Accept: application/json'
]);

$response = curl_exec($ch);
$invoice = json_decode($response, true);
curl_close($ch);
```

</CodeGroup>
